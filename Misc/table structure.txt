CREATE TABLE incidents (
        report_number VARCHAR(255),
        local_case_number VARCHAR(255),
        agency_name VARCHAR(255),
        acrs_report_type VARCHAR(255),
        crash_date_time DATETIME,
        hit_run VARCHAR(10),
        route_type VARCHAR(255),
        mile_point FLOAT,
        mile_point_direction VARCHAR(10),
        lane_direction VARCHAR(10),
        lane_number INT,
        lane_type VARCHAR(255),
        number_of_lanes INT,
        direction VARCHAR(10),
        distance FLOAT,
        distance_unit VARCHAR(10),
        road_grade VARCHAR(255),
        non_traffic VARCHAR(10),
        road_name VARCHAR(255),
        cross_street_type VARCHAR(255),
        cross_street_name VARCHAR(255),
        off_road_description TEXT,
        municipality VARCHAR(255),
        related_non_motorist VARCHAR(255),
        at_fault VARCHAR(10),
        collision_type VARCHAR(255),
        weather VARCHAR(255),
        surface_condition VARCHAR(255),
        light VARCHAR(255),
        traffic_control VARCHAR(255),
        driver_substance_abuse VARCHAR(255),
        non_motorist_substance_abuse VARCHAR(255),
        first_harmful_event VARCHAR(255),
        second_harmful_event VARCHAR(255),
        fixed_object_struck VARCHAR(255),
        junction VARCHAR(255),
        intersection_type VARCHAR(255),
        intersection_area VARCHAR(255),
        road_alignment VARCHAR(255),
        road_condition VARCHAR(255),
        road_division VARCHAR(255),
        latitude FLOAT,
        longitude FLOAT
    )
